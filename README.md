# Instructions

# install dependencies

npm install

# following env varible need to be configured before running the project

DB_USERNAME=
DB_PASSWORD=
DB_NAME=
AWS_IAM_USER_KEY=
AWS_IAM_USER_SECRET=
AWS_BUCKET_REGION=
AWS_BUCKET_NAME=

# run server
npm start

# update swagger

npm run swagger

# swagger endpoint - please check models section for request body

http://localhost:6001/api-docs/#

# to lint the project

npm run eslint-fix
